def selected_data(df,selected_clm,v1,v2):
    data = df
    #print(data)
    
    df1 = data[(data['weight'] >=v1) & (data['weight'] <=v2)]
    X1 = df1[selected_clm[0]]
    Y1 = df1[selected_clm[1]]
    r1 = scipy.stats.linregress(X1, Y1)
    print("weight range = 31.4-50")
    print("slope1: %f    intercept1: %f  R1-squared: %f" % (r1.slope, r1.intercept,r1.rvalue))
    plt.rcParams['figure.figsize'] = [15, 8]
    plt.plot(X1, Y1, 'o', label='original data')
    plt.plot(X1, r1.intercept + r1.slope*X1, 'r', label='fitted line')
    plt.xlabel('weight')
    plt.ylabel('DM Cost')
    plt.legend()
    plt.show()
    
    return 0
